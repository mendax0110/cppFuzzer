430 output << CarriageReturn;

438 if (!m_body.empty())
439 output << "Content-Length: " << m_body.size() << CarriageReturn;
440
441 // seperate headers and body with an extra carriage return
442 output << CarriageReturn;
443
444 // convert the 8-bit unsigned body to a std::string for output

591 void Parse(HTTPMessage* httpMessage, const std::vector<uint8_t>& buffer)
592 {
593 // begin by parsing the start line without knowing if it is a
594 // request or a response by setting as undetermined
595 MessageParserState state = MessageParserState::PARSING_START_LINE;
596
597 // a temporary string instance used for storing characters of a
598 // current line in the message being parsed
599 std::string temp = "";
600
601 // whether to skip the next character (for a carriage return)
602 bool skipNext = false;
603
604 // the current key for a header
605 std::string headerKey = "";
606
607 // whether or not a message body is present
608 bool hasMessageBody = false;
609
610 // the index at which the message body begins
611 size_t bodyStartIndex = 0;
612
613 for (size_t index = 0; index < buffer.size(); index++)
614 {
615 uint8_t character = buffer[index];
616
617 // skip this character as it was marked
618 if (skipNext)
619 {
620 skipNext = false;
621
622 continue;
623 }
624
625 // if we are parsing the body, then we only need to grab an index and break
626 // out of this loop as we want to merely insert the data from this vector
627 // into the body vector
628 if (state == MessageParserState::PARSING_BODY)
629 {
630 hasMessageBody = true;
631
632 bodyStartIndex = index;
633
634 break;
635 }
636
637 // if we are parsing the start line but neither a response or request
638 if (state == MessageParserState::PARSING_START_LINE)
639 {
640 // if we hit a space, we have to check if the start line begins
641 // with the HTTP version or the method verb
642 if (character == ' ')
643 {
644 // this message has a leading version string, thus it is
645 // a response and not a request
646 if (temp == HTTPVersion1 || temp == HTTPVersion11)
647 {
648 httpMessage->SetMethod(MessageMethod::NONE);
649
650 state = MessageParserState::START_LINE_RESPONSE;
651
652 temp = "";
653
654 continue;
655 }
656 // this must be a request, so grab the MessageMethod type
657 // for the request, set it, and move on
658 else
659 {
660 httpMessage->SetMethod(StringToMessageMethod(temp));
661
662 state = MessageParserState::START_LINE_REQUEST;
663
664 temp = "";
665
666 continue;
667 }
668 }
669 }
670 // do actions for when the start line is a request
671 else if (state == MessageParserState::START_LINE_REQUEST)
672 {
673 // once a space is hit, add the path to the message
674 if (character == ' ')
675 {
676 httpMessage->SetPath(temp);
677
678 temp = "";
679
680 continue;
681 }
682 // when the beginning of a carriage return is hit, add the version string
683 // to the message and then skip the following new line character, setting
684 // the state of the parser to be parsing headers
685 else if (character == '\r')
686 {
687 httpMessage->SetVersion(temp);
688
689 temp = "";
690
691 state = MessageParserState::HEADER_KEY;
692
693 skipNext = true;
694
695 continue;
696 }
697 }
698 // do actions for when the start line is a response
699 else if (state == MessageParserState::START_LINE_RESPONSE)
700 {
701 // if we are at a space, then we have hit the status code for the response
702 if (character == ' ')
703 {
704 int code = std::stoi(temp);
705
706 httpMessage->SetStatusCode(static_cast<uint16_t>(code));
707
708 temp = "";
709
710 continue;
711 }
712 // if we are at a carriage return start, then set the status message for
713 // the response, this can be blank in which it will use a generated status
714 //
715 // this will also set the state of the parser to move on to headers
716 else if (character == '\r')
717 {
718 httpMessage->SetStatusMessage(temp);
719
720 temp = "";
721
722 state = MessageParserState::HEADER_KEY;
723
724 skipNext = true;
725
726 continue;
727 }
728 }
729 // if we are parsing header keys and hit a colon, then the key for the header has
730 // been fully parsed and should be added to the temporary key holder
731 else if (state == MessageParserState::HEADER_KEY && character == ':')
732 {
733 headerKey = temp;
734
735 temp = "";
736
737 state = MessageParserState::HEADER_VALUE;
738
739 // HTTP defines that the next character in a header should be a space
740 // so skip that for parsing the value of the header
741 skipNext = true;
742
743 continue;
744 }
745 // if we are parsing header values and hit the beginning of a carriage return then
746 // it is time to add the header to the message with the key and value, and move the
747 // state back to parsing keys
748 else if (state == MessageParserState::HEADER_VALUE && character == '\r')
749 {
750 httpMessage->SetHeader(headerKey, temp);
751
752 headerKey = "";
753 temp = "";
754
755 state = MessageParserState::HEADER_KEY;
756
757 // skip the next character as it will just be a newline
758 skipNext = true;
759
760 continue;
761 }
762 // if we are parsing header keys and we hit a carriage return, then we should assume
763 // that the headers have ended, and that we are now parsing a message body.
764 else if (state == MessageParserState::HEADER_KEY && character == '\r')
765 {
766 temp = "";
767
768 state = MessageParserState::PARSING_BODY;
769
770 // skip the next character as it'll be a newline
771 skipNext = true;
772
773 continue;
774 }
775
776 temp += character;
777 }
778
779 // add the body to the message if it is present
780 if (hasMessageBody)
781 httpMessage->GetMessageBody().insert(
782 httpMessage->GetMessageBody().begin(),
783 buffer.begin() + bodyStartIndex,
784 buffer.end()
785 );
786 }
787
