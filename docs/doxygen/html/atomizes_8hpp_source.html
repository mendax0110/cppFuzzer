406 std::string ToString() const
407 {
408 std::stringstream output;
409
410 // begin by forming the start line of the message
411 if (m_method == MessageMethod::NONE)
412 {
413 output << HTTPVersion11 << " " << m_statusCode << " ";
414
415 if (m_statusMessage.empty())
416 output << StatusTextFromStatusCode(m_statusCode);
417 else
418 output << m_statusMessage;
419 }
420 else
421 {
422 output << MessageMethodToString(m_method) << " ";
423 output << m_path << " ";
424 output << HTTPVersion11;
425 }
426
427 // output the status lines line break to move on
428 output << CarriageReturn;
429
430 // output headers to the message string
431 for (auto& header : m_headers)
432 output << header.first << ": " << header.second << CarriageReturn;
433
434 // automatically output the content length based on
435 // the size of the body member if body isn't empty
436 if (!m_body.empty())
437 output << "Content-Length: " << m_body.size() << CarriageReturn;
438
439 // seperate headers and body with an extra carriage return
440 output << CarriageReturn;
441
442 // convert the 8-bit unsigned body to a std::string for output
443 std::string body(m_body.begin(), m_body.end());
444
445 output << body;
446
447 return output.str();

589 void Parse(HTTPMessage* httpMessage, const std::vector<uint8_t>& buffer)
590 {
591 // begin by parsing the start line without knowing if it is a
592 // request or a response by setting as undetermined
593 MessageParserState state = MessageParserState::PARSING_START_LINE;
594
595 // a temporary string instance used for storing characters of a
596 // current line in the message being parsed
597 std::string temp = "";
598
599 // whether to skip the next character (for a carriage return)
600 bool skipNext = false;
601
602 // the current key for a header
603 std::string headerKey = "";
604
605 // whether or not a message body is present
606 bool hasMessageBody = false;
607
608 // the index at which the message body begins
609 size_t bodyStartIndex = 0;
610
611 for (size_t index = 0; index < buffer.size(); index++)
612 {
613 uint8_t character = buffer[index];
614
615 // skip this character as it was marked
616 if (skipNext)
617 {
618 skipNext = false;
619
620 continue;
621 }
622
623 // if we are parsing the body, then we only need to grab an index and break
624 // out of this loop as we want to merely insert the data from this vector
625 // into the body vector
626 if (state == MessageParserState::PARSING_BODY)
627 {
628 hasMessageBody = true;
629
630 bodyStartIndex = index;
631
632 break;
633 }
634
635 // if we are parsing the start line but neither a response or request
636 if (state == MessageParserState::PARSING_START_LINE)
637 {
638 // if we hit a space, we have to check if the start line begins
639 // with the HTTP version or the method verb
640 if (character == ' ')
641 {
642 // this message has a leading version string, thus it is
643 // a response and not a request
644 if (temp == HTTPVersion1 || temp == HTTPVersion11)
645 {
646 httpMessage->SetMethod(MessageMethod::NONE);
647
648 state = MessageParserState::START_LINE_RESPONSE;
649
650 temp = "";
651
652 continue;
653 }
654 // this must be a request, so grab the MessageMethod type
655 // for the request, set it, and move on
656 else
657 {
658 httpMessage->SetMethod(StringToMessageMethod(temp));
659
660 state = MessageParserState::START_LINE_REQUEST;
661
662 temp = "";
663
664 continue;
665 }
666 }
667 }
668 // do actions for when the start line is a request
669 else if (state == MessageParserState::START_LINE_REQUEST)
670 {
671 // once a space is hit, add the path to the message
672 if (character == ' ')
673 {
674 httpMessage->SetPath(temp);
675
676 temp = "";
677
678 continue;
679 }
680 // when the beginning of a carriage return is hit, add the version string
681 // to the message and then skip the following new line character, setting
682 // the state of the parser to be parsing headers
683 else if (character == '\r')
684 {
685 httpMessage->SetVersion(temp);
686
687 temp = "";
688
689 state = MessageParserState::HEADER_KEY;
690
691 skipNext = true;
692
693 continue;
694 }
695 }
696 // do actions for when the start line is a response
697 else if (state == MessageParserState::START_LINE_RESPONSE)
698 {
699 // if we are at a space, then we have hit the status code for the response
700 if (character == ' ')
701 {
702 int code = std::stoi(temp);
703
704 httpMessage->SetStatusCode(static_cast<uint16_t>(code));
705
706 temp = "";
707
708 continue;
709 }
710 // if we are at a carriage return start, then set the status message for
711 // the response, this can be blank in which it will use a generated status
712 //
713 // this will also set the state of the parser to move on to headers
714 else if (character == '\r')
715 {
716 httpMessage->SetStatusMessage(temp);
717
718 temp = "";
719
720 state = MessageParserState::HEADER_KEY;
721
722 skipNext = true;
723
724 continue;
725 }
726 }
727 // if we are parsing header keys and hit a colon, then the key for the header has
728 // been fully parsed and should be added to the temporary key holder
729 else if (state == MessageParserState::HEADER_KEY && character == ':')
730 {
731 headerKey = temp;
732
733 temp = "";
734
735 state = MessageParserState::HEADER_VALUE;
736
737 // HTTP defines that the next character in a header should be a space
738 // so skip that for parsing the value of the header
739 skipNext = true;
740
741 continue;
742 }
743 // if we are parsing header values and hit the beginning of a carriage return then
744 // it is time to add the header to the message with the key and value, and move the
745 // state back to parsing keys
746 else if (state == MessageParserState::HEADER_VALUE && character == '\r')
747 {
748 httpMessage->SetHeader(headerKey, temp);
749
750 headerKey = "";
751 temp = "";
752
753 state = MessageParserState::HEADER_KEY;
754
755 // skip the next character as it will just be a newline
756 skipNext = true;
757
758 continue;
759 }
760 // if we are parsing header keys and we hit a carriage return, then we should assume
761 // that the headers have ended, and that we are now parsing a message body.
762 else if (state == MessageParserState::HEADER_KEY && character == '\r')
763 {
764 temp = "";
765
766 state = MessageParserState::PARSING_BODY;
767
768 // skip the next character as it'll be a newline
769 skipNext = true;
770
771 continue;
772 }
773
774 temp += character;
775 }
776
777 // add the body to the message if it is present
778 if (hasMessageBody)
779 httpMessage->GetMessageBody().insert(
780 httpMessage->GetMessageBody().begin(),
781 buffer.begin() + bodyStartIndex,
782 buffer.end()
783 );
784 }
